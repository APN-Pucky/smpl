import uncertainties.unumpy as unp

[7]:

rdata = pd.read_csv('test_linear_data.txt',delimiter=" ")
data = pd.DataFrame()
data['x'] = unp.uarray(rdata['x'],rdata['dx'])
data['y'] = unp.uarray(rdata['y'],rdata['dy'])
data['z'] = data['x']*data['y']
data
