def lorentz(x,x_0,A,d,y):
    '''$\\frac{A}{\\pi d (1+ (\\frac{x-x_0}{d})^2)} + y$'''
    return 1/(np.pi*d*(1+(x-x_0)**2/d**2))*A + y

    '''$A\\cdot \\exp\\left(\\frac{-(x-x_0)^2}{2d^2}\\right)+y$'''
